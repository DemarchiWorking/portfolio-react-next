import type { Metadata } from "next";
import { Inter } from "next/font/google";
import "./globals.css";
import { ThemeProvider } from "next-themes"; // Importe aqui

const inter = Inter({ subsets: ["latin"] });

export const metadata: Metadata = {
  title: "Meu Projeto Dark/Light",
  description: "Exemplo com temas",
};

export default function RootLayout({
  children,
}: Readonly<{
  children: React.ReactNode;
}>) {
  return (
    <html lang="pt-BR" suppressHydrationWarning> {/* Suprime warnings de hidratação */}
      <body className={`${inter.className} antialiased`}>
        <ThemeProvider
          attribute="class" // Usa class="dark" no <html>
          defaultTheme="system" // Respeita preferências do SO (melhor prática UX)
          enableSystem // Ativa detecção de system prefs
          disableTransitionOnChange // Opcional: Desativa flash inicial; use se preferir
        >
          {children}
        </ThemeProvider>
      </body>
    </html>
  );
}

@apply transition-colors duration-300 ease-in-out;

@layer base {
  *, *::before, *::after {
    @apply transition-colors duration-300 ease-in-out;
  }
}
